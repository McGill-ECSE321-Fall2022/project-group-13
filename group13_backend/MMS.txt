class MuseumManagementSystem
{
  Time openTime;
  Time closeTime;
  Integer maxLoanNumber;
  
  Double ticketFee;
  1 <@>- 1 Manager manager;
  1 <@>- * Employee employees;
  1 <@>- * Client clients;
  1 <@>- * Shift shifts;
  1 <@>- * LoanRequest loanRequests;
  1 <@>- * DonationRequest donationRequests;
  1 <@>- * Artifact artifacts;
  1 <@>- * Room rooms;
  1 <@>- * Ticket tickets;
  1 <@>- 0..7 SpecificWeekDay weekDays;
}

class Account
{
  abstract;
  unique username;
  name;
  password;
}

class Manager
{
  isA Account;
}

class Employee
{
  isA Account;
  
  1 employee -- * Shift shifts;
}

class Client
{
  isA Account;
  Integer currentLoanNumber;
  1 client -- * Request requests;
  1 client -- * Ticket tickets;
}

class Shift
{
  unique shiftId;
  Time startTime;
  Time endTime;

  * shifts -- 1 SpecificWeekDay dayOfTheWeek;
}

class SpecificWeekDay
{
  enum DayType { Monday, Tuesday, Wednesday, Thursday, Friday, Saturday, Sunday }
  Boolean isClosed;
  DayType dayType;
}

class Request
{
  abstract;
  unique requestId;
}

class LoanRequest
{

  enum LoanStatus { Approved, Rejected, Pending }

  isA Request;
  Integer loanDuration;
  Double fee;
  LoanStatus status;
}

class DonationRequest
{
  enum DonationStatus { Approved, Rejected, Pending, Returned}
  isA Request;
  DonationStatus status;
}

class Artifact
{
  enum LoanStatus { Available, Unavailable, Loaned }
  unique artifactId;
  name;
  image;
  description;
  LoanStatus loanStatus;
  Boolean isDamaged;
  Double loanFee;
  Double worth; 
  1 artifact -- * Request requests;
  * artifacts -- 1 Room room;
}

class Room
{
  enum RoomType { Small, Large, Storage }
  RoomType type;
  unique roomId;
}

class Ticket
{
  unique ticketId;
  Double fee;
  Boolean isActive;
}
